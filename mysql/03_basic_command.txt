
1. 创建数据库

    create database student charset=utf8mb4;

2. 创建表

    create table handsom(
        id int auto_increment primary key not null,
        name varchar(20) nut null,
        birthday datetime,
        gender bit default 1,
        isDelete bit default 0
    );

3. 插入数据

    insert into handsom values(0, 'libo', '1990-10-10', 1, 0);

4. 修改数据

    update handsom set name = 'tiger' where name = 'libo';

5. 删除数据

    delete from handsom where id = 1;
